const containerRef = useRef(null);

  initial={{ opacity: 0, y: 30 }}

>

// Individual boxes (staggered entrance)
<motion.div
  initial={{ opacity: 0, y: 40 }}

>

// CTA box (delayed)
<motion.div
  initial={{ opacity: 0, scale: 0.95 }}

>
